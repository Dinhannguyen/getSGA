local CHECK_INTERVAL = 3 -- Kiểm tra mỗi 3 giây theo yêu cầu
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local player = Players.LocalPlayer

local USERNAME = player.Name
if USERNAME == "baohan_200" then return end

repeat task.wait() until game:IsLoaded()
repeat task.wait() until player

-- Biến lưu trữ kết quả 5 lần gần nhất để xác thực
local history = {}

-- Hàm lấy thông tin item từ Inventory
local function getInventoryData()
    local success, result = pcall(function()
        local remote = ReplicatedStorage:FindFirstChild("Remotes") and ReplicatedStorage.Remotes:FindFirstChild("CommF_")
        if remote and remote.InvokeServer then
            return remote:InvokeServer("getInventory")
        end
    end)
    return (success and typeof(result) == "table") and result or {}
end

-- Hàm kiểm tra trạng thái dựa trên ưu tiên
local function checkStatus()
    local inventory = getInventoryData()
    local hasSGA = false
    local vampireCount = 0

    -- 1. Kiểm tra Sanguine Art (trong Inventory)
    for _, item in ipairs(inventory) do
        local itemName = string.lower(item.Name or "")
        if itemName:find("sanguine art") then
            hasSGA = true
        end
        if itemName:find("vampire fang") then
            vampireCount = item.Count or 0
        end
    end

    -- 2. Kiểm tra Sanguine Art (đang cầm trên tay hoặc trong Backpack)
    if not hasSGA then
        local searchLoc = {player.Character, player:FindFirstChild("Backpack")}
        for _, loc in ipairs(searchLoc) do
            if loc then
                for _, tool in ipairs(loc:GetChildren()) do
                    if tool:IsA("Tool") and string.lower(tool.Name):find("sanguine art") then
                        hasSGA = true; break
                    end
                end
            end
            if hasSGA then break end
        end
    end

    -- logic ưu tiên: SGA > Vampire Fang < 20
    if hasSGA then
        return "Completed-SGA"
    elseif vampireCount < 20 then
        return "Completed-notSGA"
    end
    
    return nil -- Không thỏa điều kiện nào
end

-- Vòng lặp chính
spawn(function()
    while true do
        local currentResult = checkStatus()

        if currentResult then
            table.insert(history, currentResult)
            
            -- Chỉ giữ lại 5 kết quả gần nhất
            if #history > 5 then
                table.remove(history, 1)
            end

            -- Kiểm tra xem 5 lần gần nhất có giống hệt nhau không
            if #history == 5 then
                local allSame = true
                for i = 1, #history do
                    if history[i] ~= currentResult then
                        allSame = false
                        break
                    end
                end

                if allSame and writefile then
                    writefile(USERNAME .. ".txt", currentResult)
                    print("Đã ghi file: " .. currentResult .. " (Xác minh 5 lần thành công)")
                end
            end
        else
            -- Nếu không thỏa điều kiện, reset lịch sử để tránh ghi nhầm
            history = {}
        end

        task.wait(CHECK_INTERVAL)
    end
end)
